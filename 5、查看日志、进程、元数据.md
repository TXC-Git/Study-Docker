# 其他常见命令

## 后台启动容器

```shell
#命令 docker run -d imageName
[root@lavm-lboagvwm5z /]# docker run -d centos
841d8412b337967e72b38d872b1e3463aeaca50d8b797ce939bc5fb585bb7ffe
[root@lavm-lboagvwm5z /]# docker ps
CONTAINER ID   IMAGE     COMMAND   CREATED   STATUS    PORTS     NAMES
[root@lavm-lboagvwm5z /]# 

#使用docker ps查看容器状态，发现容器停止了，
#常见的坑：docker后台运行容器时，必须要有一个对应的前台进程，否则docker会自动停止
```

## 查看日志

```shell

docker logs [可选项] 容器id
	-f
	-t		#添加日志产生时间
	--tail=nums #显示日志末尾的nums条日志
	
docker logs -f -t --tail=nums containerId 	#可能没有日志输入
#自己编写脚本
docker  run -d centos /bin/bash -c "while true;do echo hello,docker;sleep 1; done"
```
