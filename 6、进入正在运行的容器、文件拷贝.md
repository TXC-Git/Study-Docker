# 二、从容器内拷贝文件到主机

```shell
docker cp 容器id:容器内路径  主机路径

[root@0758501922b3 /]# exit          
exit
[root@lavm-lboagvwm5z ~]# docker cp 0758501922b3:/txcyyds.c /home/
Successfully copied 1.54kB to /home/
[root@lavm-lboagvwm5z ~]# ls
[root@lavm-lboagvwm5z ~]# cd /home
[root@lavm-lboagvwm5z home]# ls
txcyyds.c

#拷贝是一个手动过程，未来可以使用-V卷的技术，实现容器内和主机目录文件的自动同步
```
